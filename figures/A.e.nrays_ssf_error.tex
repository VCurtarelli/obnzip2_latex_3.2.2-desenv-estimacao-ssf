\NewDocumentCommand{\getlastitem}{m}{%
	\pgfplotstableread[col sep=comma]{#1}\loadedtable
	\pgfplotstablegetrowsof{\loadedtable}
	\pgfmathtruncatemacro{\lastrow}{\pgfplotsretval - 1}
	\pgfplotstablegetcolsof{\loadedtable}
	\pgfmathtruncatemacro{\lastcol}{\pgfplotsretval - 1}
	\pgfplotstablegetelem{\lastrow}{[index]\lastcol}\of{\loadedtable}
	\edef\lastitem{\pgfplotsretval}
}
