\tl_new:N \l__bold_output_string_tl

\cs_new_protected:Npn \bold_string_process:nnn #1 #2 #3
{
	\seq_set_split:Nnn \l__bold_input_string_seq { } { #3 }
	\seq_map_inline:Nn \l__bold_input_string_seq
	{ #1 { ##1 } }
	#2
}
\cs_new_protected:Npn \bold_add_accent:nn #1 #2
{
	\tl_set:Nn \l__bold_output_string_tl { \ensuremath{#1} }
	\bold_string_process:nnn
	{ \bold_iterate_accent:n }
	{ \tl_use:N \l__bold_output_string_tl }
	{ #2 }
}
\cs_new_protected:Npn \bold_iterate_accent:n #1
{
	\tl_set:Nx \l__bold_output_string_tl
	{
		% \exp_args:No 
		\use:c { @my#1 }
		{ \tl_use:N \l__bold_output_string_tl }
	}
}

}

\DeclareDocumentCommand{\bv}{m O{n} g}{
	\@myaccent{\symbfup{#1}}{#2}\IfValueT{#3}{_{#3}}
}
\DeclareDocumentCommand{\BV}{m g}{
	\symbfsfup{#1}\IfValueT{#2}{_{#2}}
}
\DeclareDocumentCommand{\newbv}{s o m}{
	\IfBooleanTF{#1}{
		\IfValueTF{#2}{
			\@namedef{bv#2}{\BV{#3}}
		}{
			\@namedef{bv#3}{\BV{#3}}
		}
	}{
		\IfValueTF{#2}{
			\@namedef{bv#2}{\bv{#3}}
		}{
			\@namedef{bv#3}{\bv{#3}}
		}
	}
}
\NewDocumentCommand{\mathbfcal}{m}{\boldsymbol{\mathcal{#1}}}

}

	\IfValueTF{#1}{
		\@namedef{cal#1}{\calbv{#2}}
	}{
		\@namedef{cal#2}{\calbv{#2}}
	}
}
